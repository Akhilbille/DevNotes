What is React + TypeScript?

React in a Nutshell

React is a declarative, component-based JavaScript library for building user interfaces. It was created by Facebook (Meta) and focuses on building UIs from independent, reusable components.

🧩 Component-Based

Every piece of your UI is a self-contained component. Components can nest, reuse, and communicate via props.

⚡ Virtual DOM

React maintains a virtual copy of the DOM. On state changes, it diffs and patches only what changed — making updates fast.

🔄 Declarative

You describe what the UI should look like for a given state. React handles how to make it happen.

📦 Unidirectional

Data flows one way — from parent to child via props. This makes data flow predictable and easy to debug.

Why TypeScript?

TypeScript adds static types to JavaScript. This means you catch errors at compile time instead of runtime — and get amazing IDE autocomplete.

📄 Without TypeScript (JavaScript) 
function greet(user) {
  return `Hello, ${user.nam}`; // typo: 'nam' instead of 'name' — no warning!
}

greet({ name: "Alice" }); // returns "Hello, undefined" silently
📄 With TypeScript 
interface User {
  name: string;
  age: number;
}

function greet(user: User): string {
  return `Hello, ${user.nam}`; // ❌ TS Error: Property 'nam' does not exist
}

greet({ name: "Alice", age: 30 }); // ✅ Correct

Your First Component (TSX)

Anatomy of a React TSX Component

A React component in TypeScript is a function that returns JSX (written in .tsx files). TypeScript lets us strongly-type props using interface or type.

src/components/ListGroup/ListGroup.tsx 
// ✅ No need to import React in modern React (17+) + Vite

// 1. Define the shape of props using an interface
interface ListGroupProps {
  items: string[];
  heading: string;
  onSelectItem: (item: string) => void;
}

// 2. Component function typed with FC (FunctionComponent) or inline return type
function ListGroup({ items, heading, onSelectItem }: ListGroupProps) {
  // JSX is just syntactic sugar for React.createElement()
  return (
    <div>
      <h1>{heading}</h1>
      {items.length === 0 ? (
        <p>No items found</p>
      ) : (
        <ul className="list-group">
          {items.map((item) => (
            <li
              key={item}
              onClick={() => onSelectItem(item)}
            >
              {item}
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}

export default ListGroup;

Fragments

React components must return a single root element. Fragments let you group elements without adding an extra DOM node.

src/components/UserInfo.tsx 
// ❌ Error: Adjacent JSX elements must be wrapped
function Bad() {
  return (
    <h1>Title</h1>
    <p>Paragraph</p>
  );
}

// ✅ Option 1: Wrap in a div (adds extra DOM node)
function WithDiv() {
  return (
    <div>
      <h1>Title</h1>
      <p>Paragraph</p>
    </div>
  );
}

// ✅ Option 2: React.Fragment (verbose)
import { Fragment } from "react";

function WithFragment() {
  return (
    <Fragment>
      <h1>Title</h1>
      <p>Paragraph</p>
    </Fragment>
  );
}

// ✅ Option 3: Short syntax <></> (most common)
function WithShortFragment() {
  return (
    <>
      <h1>Title</h1>
      <p>Paragraph</p>
    </>
  );
}

// ✅ When you need a key prop (e.g., in a map), use Fragment explicitly

    <>
      {items.map((item, idx) => (
        <Fragment key={idx}>
          <dt>{item}</dt>
          <dd>A delicious fruit</dd>
        </Fragment>
      ))}
    </>
  );
}

Rendering Lists

Use Array.map() to transform data arrays into JSX elements. Every element in a list needs a unique key prop.

src/components/ProductList.tsx 
// Define the data shape
interface Product {
  id: number;
  name: string;
  price: number;
  category: string;
  inStock: boolean;
}

interface ProductListProps {
  products: Product[];
}

function ProductList({ products }: ProductListProps) {
  // Guard: nothing to display
  if (products.length === 0) {
    return <p>No products available</p>;
  }

  return (
    <ul>
      {products.map((product: Product) => (
        <li key={product.id}>     {/* key must be unique & stable */}
          <strong>{product.name}</strong> — ${product.price}
          {!product.inStock && <span> (Out of stock)</span>}
        </li>
      ))}
    </ul>
  );
}

// Usage in App.tsx
const products: Product[] = [
  { id: 1, name: "Laptop", price: 999, category: "Tech", inStock: true },
  { id: 2, name: "Mouse",  price: 29,  category: "Tech", inStock: false },
  { id: 3, name: "Desk",   price: 450, category: "Furniture", inStock: true },
];
⚠️
Key Rules for the key Prop

Never use array index as key if the list can be reordered or filtered. Use a stable, unique ID from your data (id from a database is ideal). Keys help React reconcile changes in lists efficiently.

Conditional Rendering

React lets you conditionally include JSX using standard JavaScript techniques — ternary operators, logical AND, or early returns.

src/components/AuthStatus.tsx 
interface AuthStatusProps {
  isLoggedIn: boolean;
  isAdmin: boolean;
  username?: string;
  errorMessage?: string;
}

function AuthStatus({ isLoggedIn, isAdmin, username, errorMessage }: AuthStatusProps) {
  // Technique 1: Early return (guard clause)
  if (errorMessage) {
    return <div className="error">{errorMessage}</div>;
  }

  return (
    <div>
      {/* Technique 2: Ternary */}
      {isLoggedIn ? (
        <p>Welcome back, {username}!</p>
      ) : (
        <p>Please log in</p>
      )}

      {/* Technique 3: Logical AND (renders nothing if false) */}
      {isAdmin && <button>Admin Panel</button>}

      {/* Technique 4: Logical OR (fallback value) */}
      <p>User: {username || "Anonymous"}</p>

      {/* Technique 5: Nullish coalescing (only null/undefined, not 0 or "") */}
      <p>Display: {username ?? "Guest"}</p>
    </div>
  );
}

// ⚠️ Gotcha: && with numbers — 0 renders as "0"!
const count = 0;
// ❌ Bad: renders "0" in the DOM
const bad = count && <span>Has items</span>;

// ✅ Good: convert to boolean first
const good = count > 0 && <span>Has items</span>;
const good2 = !!count && <span>Has items</span>;

Handling Events

React's synthetic event system wraps native browser events. TypeScript requires you to specify the correct event type — which gives you full autocomplete on the event object.

src/components/EventDemo.tsx 
import { MouseEvent, ChangeEvent, FormEvent, KeyboardEvent } from "react";

function EventDemo() {
  // MouseEvent handler
  const handleClick = (e: MouseEvent<HTMLButtonElement>): void => {
    console.log("Clicked at:", e.clientX, e.clientY);
    e.preventDefault(); // stops default browser behavior
  };

  // ChangeEvent handler — for inputs
  const handleChange = (e: ChangeEvent<HTMLInputElement>): void => {
    console.log("Input value:", e.target.value);
  };

  // Select element change
  const handleSelectChange = (e: ChangeEvent<HTMLSelectElement>): void => {
    console.log("Selected:", e.target.value);
  };

  // Keyboard event
  const handleKeyDown = (e: KeyboardEvent<HTMLInputElement>): void => {
    if (e.key === "Enter") console.log("Enter pressed!");
    if (e.key === "Escape") console.log("Escape pressed!");
  };

  // Form submit
  const handleSubmit = (e: FormEvent<HTMLFormElement>): void => {
    e.preventDefault(); // prevent page reload!
    console.log("Form submitted");
  };

  return (
    <>
      <button onClick={handleClick}>Click Me</button>
      <input onChange={handleChange} onKeyDown={handleKeyDown} />
      <select onChange={handleSelectChange}>
        <option value="a">Option A</option>
      </select>
      <form onSubmit={handleSubmit}>
        <button type="submit">Submit</button>
      </form>
    </>
  );
}

onChangeChangeEvent<HTMLInputElement>Inputs, textareas
onChange (select)ChangeEvent<HTMLSelectElement>Dropdowns

Passing Data Via Props

src/components/Alert.tsx 
import { ReactNode } from "react";

// children is typed as ReactNode — can be text, JSX, or null
interface AlertProps {
  type: "info" | "warning" | "danger" | "success";
  title: string;
  children: ReactNode;        // anything React can render
  onClose?: () => void;        // optional callback
  dismissible?: boolean;       // optional boolean, default false
}

function Alert({
  type,
  title,
  children,
  onClose,
  dismissible = false,           // default parameter
}: AlertProps) {
  const colorMap: Record<AlertProps["type"], string> = {
    info:    "alert-info",
    warning: "alert-warning",
    danger:  "alert-danger",
    success: "alert-success",
  };

  return (
    <div className={`alert ${colorMap[type]}`}>
      <strong>{title}</strong>
      <div>{children}</div>
      {dismissible && (
        <button onClick={onClose}>✕</button>
      )}
    </div>
  );
}

// Usage — children is whatever you put between tags
function App() {
  return (
    <Alert type="warning" title="Warning!" dismissible onClose={() => alert("closed")}>
      <p>Your session is about to expire.</p>
      <a href="/refresh">Refresh now</a>
    </Alert>
  );
}

CSS & CSS Modules

Vanilla CSS

Import a global CSS file directly. Styles are global — may cause naming conflicts in large apps.

src/components/Button/Button.tsx 
import "./Button.css"; // global CSS

interface ButtonProps {
  label: string;
  variant: "primary" | "secondary" | "danger";
  size?: "sm" | "md" | "lg";
  disabled?: boolean;
  onClick: () => void;
}

function Button({ label, variant, size = "md", disabled, onClick }: ButtonProps) {
  // Dynamically build className string
  const classes = ["btn", `btn-${variant}`, `btn-${size}`]
    .filter(Boolean)
    .join(" ");

  return (
    <button
      className={classes}
      disabled={disabled}
      onClick={onClick}
    >
      {label}
    </button>
  );
}

useState Hook

State is data that can change over time. When state changes, React re-renders the component with the new value. useState is the primary hook for managing local component state.

src/components/StateExamples.tsx 
import { useState } from "react";

// Basic useState — TypeScript infers the type from initial value
function Counter() {
  const [count, setCount] = useState(0); // inferred: number

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>+</button>
      <button onClick={() => setCount(prev => prev - 1)}>-</button>  // updater fn pattern
    </div>
  );
}

// Explicit type annotation — needed when initial value is null/undefined
interface User { id: number; name: string; }

function UserProfile() {
  const [user, setUser] = useState<User | null>(null); // explicit generic type
  const [error, setError] = useState<string>("");
  const [isLoading, setIsLoading] = useState<boolean>(false);

  return (
    <div>
      {isLoading && <p>Loading...</p>}
      {error && <p style={{ color: "red" }}>{error}</p>}
      {user && <p>Welcome, {user.name}</p>}
    </div>
  );
}

// useState with union type
type Theme = "light" | "dark" | "auto";

function ThemeToggle() {
  const [theme, setTheme] = useState<Theme>("light");

  const cycleTheme = (): void => {
    const themes: Theme[] = ["light", "dark", "auto"];
    const nextIndex = (themes.indexOf(theme) + 1) % themes.length;
    setTheme(themes[nextIndex]);
  };

  return <button onClick={cycleTheme}>Theme: {theme}</button>;
}

// ⚠️ IMPORTANT: Keep state pure — never mutate directly
// ❌ Wrong: count++  or  user.name = "Bob"
// ✅ Right: setCount(count + 1)  or  setUser({ ...user, name: "Bob" })

Updating Objects & Arrays in State

State must be treated as immutable. You never mutate state directly — you always create a new object/array.

src/components/ImmutableUpdates.tsx 
import { useState } from "react";

interface Address { street: string; city: string; zip: string; }
interface Person  { name: string; age: number; address: Address; }

function PersonEditor() {
  const [person, setPerson] = useState<Person>({
    name: "Alice",
    age: 30,
    address: { street: "123 Main St", city: "NYC", zip: "10001" }
  });

  // ❌ WRONG — mutates state directly, React won't re-render
  const badUpdate = () => {
    person.name = "Bob"; // ❌ mutation!
    setPerson(person);  // ❌ same reference — React ignores this
  };

  // ✅ Updating a flat object — spread operator
  const updateName = (name: string): void => {
    setPerson({ ...person, name }); // new object with same props + updated name
  };

  // ✅ Updating a NESTED object — spread at each level
  const updateCity = (city: string): void => {
    setPerson({
      ...person,
      address: { ...person.address, city }, // spread nested object too
    });
  };

  // ============================
  // ARRAY STATE OPERATIONS
  // ============================

  const [tags, setTags] = useState<string[]>(["react", "typescript"]);

  // ADD — spread + new item
  const addTag = (tag: string): void => {
    setTags([...tags, tag]);
  };

  // REMOVE — filter out the item
  const removeTag = (tagToRemove: string): void => {
    setTags(tags.filter(t => t !== tagToRemove));
  };

  // UPDATE — map over and replace the matching item
  interface Todo { id: number; text: string; done: boolean; }
  const [todos, setTodos] = useState<Todo[]>([]);

  const toggleTodo = (id: number): void => {
    setTodos(todos.map(todo =>
      todo.id === id ? { ...todo, done: !todo.done } : todo
    ));
  };

  // SORT — always create a new array (sort mutates!)

      <div>
        <input
          name="email"
          type="email"
          value={formData.email}
          onChange={handleChange}

Schema-Based Validation with Zod

Zod lets you define a validation schema, then infer the TypeScript type from it. Combining Zod with RHF gives you the cleanest form validation in React.

src/components/ExpenseForm.tsx 
// Install: npm install zod @hookform/resolvers
import { z } from "zod";
import { zodResolver } from "@hookform/resolvers/zod";
import { useForm, SubmitHandler } from "react-hook-form";

// 1. Define schema with Zod
const expenseSchema = z.object({
  description: z.string()
    .min(3, "Description must be at least 3 chars")
    .max(50, "Description too long"),

  amount: z.number({ invalid_type_error: "Amount must be a number" })
    .positive("Amount must be positive")
    .max(100_000, "Too large"),

  category: z.enum(["groceries", "utilities", "entertainment", "other"]),

  date: z.string().refine((val) => !isNaN(Date.parse(val)), {
    message: "Invalid date",
  }),
});

// 2. Infer TypeScript type FROM the schema (no duplication!)
type ExpenseFormData = z.infer<typeof expenseSchema>;

// Result:
// type ExpenseFormData = {
//   description: string;
//   amount: number;
//   category: "groceries" | "utilities" | "entertainment" | "other";
//   date: string;
// }

function ExpenseForm() {
  const {
    register,
    handleSubmit,
    formState: { errors },
    reset,
  } = useForm<ExpenseFormData>({
    resolver: zodResolver(expenseSchema), // connect Zod schema to RHF
  });

  const onSubmit: SubmitHandler<ExpenseFormData> = (data) => {
    console.log("Valid expense:", data); // only called if schema passes!
    reset();
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input {..}register("description") placeholder="Description" />
      {errors.description && <p>{errors.description.message}</p>}

      <input type="number" {..}register("amount", { valueAsNumber: true }) placeholder="Amount" />
      {errors.amount && <p>{errors.amount.message}</p>}

useEffect Hook

useEffect lets you synchronize your component with external systems — APIs, timers, subscriptions, the document title, etc. It runs after the component renders.

src/components/EffectExamples.tsx 
import { useState, useEffect } from "react";

function EffectExamples() {
  const [count, setCount] = useState(0);
  const [userId, setUserId] = useState(1);

  // 1. Runs after EVERY render (no dependency array)
  useEffect(() => {
    console.log("Rendered!");
  }); // ⚠️ usually avoid this pattern — too frequent

  // 2. Runs ONCE — on mount (empty dependency array)
  useEffect(() => {
    console.log("Component mounted");
    document.title = "My App";
    // Cleanup — runs on unmount
    return () => {
      document.title = "React App"; // restore on unmount
    };
  }, []); // empty array = run once

  // 3. Runs when specific dependency changes
  useEffect(() => {
    document.title = `Count: ${count}`;
  }, [count]); // only re-run when count changes

  // 4. Cleanup for subscriptions / timers
  useEffect(() => {
    const interval = setInterval(() => {
      console.log("Tick");
    }, 1000);

    return () => clearInterval(interval); // cleanup prevents memory leak
  }, []);

  // 5. With AbortController to cancel fetch on unmount/dependency change
  useEffect(() => {
    const controller = new AbortController();

    async function fetchUser() {
      try {
        const res = await fetch(`/api/users/${userId}`, {
          signal: controller.signal,
        });
        const data = await res.json();
        console.log("User:", data);
      } catch (err) {
        if (err instanceof Error && err.name === "AbortError") {
          console.log("Fetch cancelled"); // normal, not an error
        }
      }
    }

    fetchUser();
    return () => controller.abort(); // cancel on cleanup
  }, [userId]);

  return (
    <button onClick={() => setCount(c => c + 1)}>Count: {count}</button>
  );
}

export interface User {
  id: number;
  name: string;
  email: string;
  username: string;
}

// Generic HTTP service class
class HttpService<T> {
  constructor(private endpoint: string) {}

  getAll() {
    const controller = new AbortController();
    const request = apiClient.get<T[]>(this.endpoint, {
      signal: controller.signal,
    });
    return { request, cancel: () => controller.abort() };
  }

  get(id: number) {
    return apiClient.get<T>(`${this.endpoint}/${id}`);
  }

  create(data: T) {
    return apiClient.post<T>(this.endpoint, data);
  }

  update(id: number, data: Partial<T>) {
    return apiClient.patch<T>(`${this.endpoint}/${id}`, data);
  }

  delete(id: number) {
    return apiClient.delete(`${this.endpoint}/${id}`);
  }
}

import { useState, useEffect } from "react";
import { userService, CanceledError, User } from "../services/userService";

function UserManager() {
  const [users, setUsers] = useState<User[]>([]);

  // ── CREATE ──────────────────────────────────────────
  const addUser = async () => {
    const newUser: User = { id: 0, name: "New User", email: "new@email.com", username: "newuser" };

    // Optimistic update — update UI first, then sync with server
    const originalUsers = [...users];
    setUsers([newUser, ...users]);

    try {
      const { data: savedUser } = await userService.create(newUser);
      setUsers([savedUser, ...users]); // use server response (has real ID)
    } catch (err) {
      setUsers(originalUsers);  // rollback on error
      console.error("Create failed", err);
    }
  };

  // ── UPDATE ──────────────────────────────────────────
  const updateUser = async (user: User) => {
    const updatedUser: User = { ...user, name: user.name + " (updated)" };
    const originalUsers = [...users];
    setUsers(users.map(u => u.id === user.id ? updatedUser : u));

    try {
      await userService.update(user.id, updatedUser);
    } catch (err) {
      setUsers(originalUsers);
      console.error("Update failed", err);
    }
  };

  // ── DELETE ──────────────────────────────────────────
  const deleteUser = async (userId: number) => {
    const originalUsers = [...users];
    setUsers(users.filter(u => u.id !== userId));

    try {
      await userService.delete(userId);
    } catch (err) {
      setUsers(originalUsers);
      console.error("Delete failed", err);
    }
  };

  return (
    <div>
      <button onClick={addUser}>Add User</button>
      <ul>
        {users.map((user) => (
          <li key={user.id}>
            {user.name}
            <button onClick={() => updateUser(user)}>Edit</button>
            <button onClick={() => deleteUser(user.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

import { useState } from "react";

// Generic custom hook with TypeScript generics
function useLocalStorage<T>(key: string, initialValue: T): [T, (val: T) => void] {
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      const item = localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch {
      return initialValue;
    }
  });

  const setValue = (value: T): void => {
    try {
      setStoredValue(value);
      localStorage.setItem(key, JSON.stringify(value));
    } catch (error) {
      console.error("LocalStorage error:", error);
    }
  };

  return [storedValue, setValue];
}

// Usage example:
function SettingsPage() {
  const [theme, setTheme] = useLocalStorage<"light" | "dark">("theme", "light");
  const [fontSize, setFontSize] = useLocalStorage<number>("fontSize", 16);

  return (
    <button onClick={() => setTheme(theme === "light" ? "dark" : "light")}>
      Toggle: {theme}
    </button>
  );
}

import { useState, useEffect } from "react";

// Delays updating value until user stops typing
function useDebounce<T>(value: T, delay: number = 500): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);

  return debouncedValue;
}

// Usage: search that only fires after user stops typing 500ms
function SearchBar() {
  const [query, setQuery] = useState("");
  const debouncedQuery = useDebounce(query, 500);

  useEffect(() => {
    if (debouncedQuery) {
      console.log("Searching for:", debouncedQuery); // API call here
    }
  }, [debouncedQuery]);

  return (
    <input
      value={query}
      onChange={(e) => setQuery(e.target.value)}
      placeholder="Search..."
    />
  );
}
